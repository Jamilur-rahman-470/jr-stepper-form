## Install

```bash
npm install --save jr-stepper-form
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'jr-stepper-form'
import 'jr-stepper-form/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
